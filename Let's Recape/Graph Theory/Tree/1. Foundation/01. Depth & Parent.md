# Depth and child count of i node : 
## DFS 
```cpp
#include<bits/stdc++.h>
using namespace std ;

const int mx = 1e3+12;
vector<int>g[mx];
int parent[mx];
int depth[mx];

void dfs( int u , int p ){
    parent[u] = p ;
    depth[u] = depth[p]+1 ;
    if (p == -1 ){
        depth[u] = 0 ;
    }

    for ( auto v : g[u]){
        if ( v == p ) continue;
        dfs(v,u);
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n ; cin >> n ;
    for ( int i = 0 ; i < n-1 ; i++ ){
        int u , v ;
        cin >> u >> v ;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    int root = 1; // assume root 
    dfs(root , -1);
    cout << parent[1] << " " << depth[1] << '\n';
    
    return 0 ;
}
```

## BFS
```cpp
#include <bits/stdc++.h>
using namespace std;

const int mx = 1e3 + 12;
vector<int> g[mx];
int parent[mx];
int depth[mx];

void bfs(int root) {
    queue<int> q;
    vector<bool> vis(mx, false);

    q.push(root);
    vis[root] = true;
    parent[root] = -1;
    depth[root] = 0;

    while (!q.empty()) {
        int u = q.front(); q.pop();

        for (auto v : g[u]) {
            if (!vis[v]) {
                vis[v] = true;
                parent[v] = u;
                depth[v] = depth[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    int root = 1; // assume root
    bfs(root);

    cout << parent[1] << " " << depth[1] << '\n';

    return 0;
}
```
