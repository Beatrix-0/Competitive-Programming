# Dead Subtree Trick

The dead subtree trick is used in tree problems where:

1. Some subtrees cannot contribute to a desired solution (like sums, colors, or counts).
2. You want to ignore them early to avoid unnecessary calculations.
3. Usually, we mark a subtree as "dead" if it violates some property, and we stop DFS for that subtree.

### Classic Problem :

Given a tree with n nodes, each node has a value a[i].
Find the maximum sum of nodes such that no two nodes in the sum are connected.
(Dead subtree trick helps ignore subtrees where sum becomes negative.)

```cpp
#include <bits/stdc++.h>
using namespace std;

#ifndef ONLINE_JUDGE
#include "trace.cpp"
#else
#define dbg(...)
#endif

const int mx = 1e5 + 5;
vector<int> g[mx];
int a[mx];
bool dead[mx];

int dfs(int u, int p) {
    int sum = a[u];

    for (int v : g[u]) {
        if (v == p) continue;
        sum += dfs(v, u);
    }

    if (sum < 0) { // dead subtree condition
        dead[u] = true;
        return 0; // ignore this subtree
    }

    return sum;
}

void solve() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    int ans = dfs(1, 0);
    cout << ans << '\n';

    cout << "Dead nodes: ";
    for (int i = 1; i <= n; i++) {
        if (dead[i]) {
            cout << i << " ";
        }
    }
    cout << "\n";
}

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int tests = 1;
    // cin >> tests;
    for ( int tc = 1 ; tc <= tests ; tc++ ){
        solve();
    }
    return 0;
}
```
