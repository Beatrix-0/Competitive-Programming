# Euler Tour Technique (ETT)

The Euler Tour Technique (ETT) is a powerful way to convert a tree into a array, so that we can apply array-based algorithms (like segment tree, Fenwick tree, etc.) on trees.

We can perform <b>2 types</b> of query and update : </br>

1. Subtree Query
2. Path Query

## Subtree Query

[Problem : Subtree Queries (CSES)](https://cses.fi/problemset/task/1137)

```cpp
#include <bits/stdc++.h>
using namespace std;

#ifndef ONLINE_JUDGE
#include "trace.cpp"
#else
#define dbg(...)
#endif

#define int long long
const int N = 2e5 + 12;

vector<int> g[N];
vector<int> val;
int subtree[N], in[N];
vector<int> flatten;

int timer = 1;
void dfs(int u, int p){
    in[u] = timer++;
    flatten.push_back(u);
    subtree[u] = 1;

    for (auto v : g[u]){
        if (v == p) continue;
        dfs(v, u);
        subtree[u] += subtree[v];
    }
}

struct ST {
    int t[4 * N];

    ST() {
        memset(t, 0, sizeof t);
    }

    inline int combine(int a, int b){
        return a + b;
    }

    void build(int n, int b, int e){
        if (b == e){
            t[n] = val[flatten[b]];
            return;
        }
        int mid = (b + e) >> 1;
        int l = n << 1;
        int r = l + 1;
        build(l, b, mid);
        build(r, mid + 1, e);
        t[n] = combine(t[l], t[r]);
    }

    void upd(int n, int b, int e, int i, int x){
        if (i < b || i > e) return;

        if (b == e){
            t[n] = x;
            return;
        }

        int mid = (b + e) >> 1;
        int l = n << 1;
        int r = l + 1;

        upd(l, b, mid, i, x);
        upd(r, mid + 1, e, i, x);

        t[n] = combine(t[l], t[r]);
    }

    int query(int n, int b, int e, int i, int j){
        if (j < b || i > e) return 0;

        if (i <= b && e <= j){
            return t[n];
        }

        int mid = (b + e) >> 1;
        int l = n << 1;
        int r = l + 1;

        return combine( query(l, b, mid, i, j), query(r, mid + 1, e, i, j));
    }

} t;

void solve() {
    int n, q;
    cin >> n >> q;

    val.resize(n + 1);
    for (int i = 1; i <= n; i++){
        cin >> val[i];
    }

    for (int i = 0; i < n - 1; i++){
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    flatten.push_back(0);
    dfs(1, -1);

    t.build(1, 1, n);
    while (q--){
        int type; cin >> type;
        if (type == 1){
            int s, x;
            cin >> s >> x;

            t.upd(1, 1, n, in[s], x);
            val[s] = x;
        }
        else{
            int s; cin >> s;
            int l = in[s];
            int r = in[s] + subtree[s] - 1;
            cout << t.query(1, 1, n, l, r) << '\n';
        }
    }
}

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int tests = 1;
    // cin >> tests;
    for ( int tc = 1 ; tc <= tests ; tc++ ){
        solve();
    }
    return 0;
}

```

## Path Query

[Problem : Path Queries (CSES)](https://cses.fi/problemset/task/1138)

```cpp
#include <bits/stdc++.h>
using namespace std;

#ifndef ONLINE_JUDGE
#include "trace.cpp"
#else
#define dbg(...)
#endif
#define int long long

const int N = 2e5+12;
vector<int>g[N];
int in[N] , out[N];
vector<int>eulerPath , val;
int cnt = 1 ;

void dfs(int u , int p ){
    in[u] = cnt;
    eulerPath.push_back(val[u]);
    cnt++;
    for ( auto v : g[u]){
        if ( v == p ) continue;
        dfs(v , u);
    }
    out[u] = cnt;
    eulerPath.push_back(-val[u]);
    cnt++;
}

struct ST {
    int t[2*4 * N];

    ST() {
        memset(t, 0, sizeof t);
    }

    inline int combine(int a, int b){
        return a + b;
    }

    void build(int n, int b, int e){
        if (b == e){
            t[n] = eulerPath[b];
            return;
        }
        int mid = (b + e) >> 1;
        int l = n << 1;
        int r = l + 1;
        build(l, b, mid);
        build(r, mid + 1, e);
        t[n] = combine(t[l], t[r]);
    }

    void upd(int n, int b, int e, int i, int x){
        if (i < b || i > e) return;

        if (b == e){
            t[n] = x;
            return;
        }

        int mid = (b + e) >> 1;
        int l = n << 1;
        int r = l + 1;

        upd(l, b, mid, i, x);
        upd(r, mid + 1, e, i, x);

        t[n] = combine(t[l], t[r]);
    }

    int query(int n, int b, int e, int i, int j){
        if (j < b || i > e) return 0;

        if (i <= b && e <= j){
            return t[n];
        }

        int mid = (b + e) >> 1;
        int l = n << 1;
        int r = l + 1;

        return combine( query(l, b, mid, i, j), query(r, mid + 1, e, i, j));
    }

} t;

void solve() {
    int n , q ;
    cin >> n >> q ;
    val.resize(n+1);
    for ( int i = 1 ; i <= n ; i++ ){
        cin >> val[i];
    }
    for ( int i = 0 ; i < n-1 ; i++ ){
        int u , v ;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    eulerPath.push_back(0); // 1 based
    dfs(1,-1);

    t.build(1, 1, 2*n);
    for( int i = 0 ; i < q ; i++ ){
        int type ; cin >> type ;
        if( type == 1 ){
            int s , x ;
            cin >> s >> x ;
            int ind1 = in[s]; // start node
            int ind2 = out[s]; // end node
            t.upd(1,1,2*n, ind1 , x);
            t.upd(1,1,2*n, ind2 , -x);
        }
        else{
            int s ; cin >> s ;
            int l = 1 , r = in[s];
            cout << t.query(1,1,2*n , l , r ) << '\n';
        }
    }
}

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int tests = 1;
    // cin >> tests;
    for ( int tc = 1 ; tc <= tests ; tc++ ){
        solve();
    }
    return 0;
}
```
