# Dijkstra O(ElogV)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int mx = 2e5 + 123;
const long long INF = 1e17;
vector<pair<int, int>> g[mx];
int dis[mx];
int n, m;

void dijkstra(int s)
{
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, s});
    dis[s] = 0;

    while (!pq.empty()){
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dis[u]) continue;
        for (auto it : g[u]){
            auto v = it.first ;
            auto w = it.second ;
            if (d + w < dis[v]){
                dis[v] = d + w;
                pq.push({dis[v], v});
            }
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++){
        int u, v, c;
        cin >> u >> v >> c;
        g[u].push_back({v, c});
    }

    fill(dis, dis + n + 1, INF);
    dijkstra(1);

    for (int i = 1; i <= n; i++){
        cout << dis[i] << " ";
    }
    cout << '\n';
    return 0;
}
```

# Dijkstra using 2 state

## Practice problem

[E. Shortest Path](https://codeforces.com/contest/59/problem/E)

```cpp
#include <bits/stdc++.h>
using namespace std;

#ifndef ONLINE_JUDGE
#include "trace.cpp"
#else
#define dbg(...)
#endif
#define int long long

const int mx = 3000+12;
const int inf = 1e17+12;
vector<int>g[mx];

void solve() {
    int n , m , k ;
    cin >> n >> m >> k ;
    for ( int i = 0 ; i < m ; i++ ){
        int u , v ;
        cin >> u >> v ;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    set<tuple<int,int,int>>bad;
    for ( int i = 0 ; i < k ; i++ ){
        int a , b , c ;
        cin >> a >> b >> c ;
        bad.insert({a,b,c});
    }

    // dis[node][prev]
    vector<vector<int>>dis(n+1, vector<int>(n+1, inf));
    // parent[node][prev] = (prev, prev_prev)
    vector<vector<pair<int,int>>>p(n+1, vector<pair<int,int>>(n+1, {-1,-1}));

    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<tuple<int, int, int>>> pq;
    pq.push({0,1,0}); // (dis,curNode,prvNode)
    dis[1][0] = 0;
    while(!pq.empty()){
        auto[d,u,prv] = pq.top();
        pq.pop();
        if ( d != dis[u][prv]) continue;

        for ( auto v : g[u]){
            if (bad.find({prv,u,v}) == bad.end()){
                if ( d+1 < dis[v][u]){
                    dis[v][u] = d+1;
                    p[v][u] = {u , prv};
                    pq.push({dis[v][u] , v , u});
                }
            }
        }
    }

    int bestPrv = -1, tmp = inf;
    for ( int i = 1 ; i <= n ; i++ ){
        if ( dis[n][i] < tmp){
            tmp = dis[n][i];
            bestPrv = i;
        }
    }

    if ( tmp == inf ){
        cout << -1 << '\n';
        return;
    }

    vector<int>path;
    int cur = n , prv = bestPrv;
    while ( cur != -1 ){
        path.push_back(cur);
        auto[nCur, nPrv] = p[cur][prv];
        cur = nCur;
        prv = nPrv;
    }
    reverse(path.begin() , path.end());
    cout << path.size()-1 << '\n';
    for (auto it : path ){
        cout << it << " ";
    }
    cout << '\n';
}

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int tests = 1;
    // cin >> tests;
    for ( int tc = 1 ; tc <= tests ; tc++ ){
        solve();
    }
    return 0;
}
```
