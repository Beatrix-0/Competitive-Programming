Code 01 : Dijkstra with priority_queue(min heap)
Time Compexity : O(ElogV)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

const int mx = 2e5 + 123;
const long long INF = 1e17;
vector<pair<int, int>> g[mx];
int dis[mx];
int n, m;

void dijkstra(int s)
{
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, s});
    dis[s] = 0;

    while (!pq.empty()){
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dis[u]) continue;
        for (auto it : g[u]){
            auto v = it.first ;
            auto w = it.second ;
            if (d + w < dis[v]){
                dis[v] = d + w;
                pq.push({dis[v], v});
            }
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++){
        int u, v, c;
        cin >> u >> v >> c;
        g[u].push_back({v, c});
    }

    fill(dis, dis + n + 1, INF);
    dijkstra(1);

    for (int i = 1; i <= n; i++){
        cout << dis[i] << " ";
    }
    cout << '\n';
    return 0;
}
```
