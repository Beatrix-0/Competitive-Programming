## DSU Code

```cpp

#include<bits/stdc++.h>
using namespace std ;

const int mx = 2e5+123;
struct Disjoint_set_union{
    int Parent[mx], Size[mx];
    void Init(int n){
        for (int i = 0; i <= n; i++){
            Parent[i] = i;
            Size[i] = 1;
        }
    }
    int Find(int v){
        if (v == Parent[v]) return v;
        return Parent[v] = Find(Parent[v]);
    }
    bool Union(int u, int v){
        int root_u = Find(u);
        int root_v = Find(v);
        if (root_u != root_v){
            if (Size[root_u] < Size[root_v]){
                swap(root_u, root_v);
            }
            Parent[root_v] = root_u;
            Size[root_u] += Size[root_v];
            return true;
        }
        return false;
    }
} dsu;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n , m ;
    cin>> n >> m ;
    dsu.Init(n);
    while ( m-- ){
        int u , v ;
        cin>> u >> v ;
        dsu.Union( u , v );
    }

    int conCmp = 0 ;
    for ( int i = 1 ; i <= n ; i++ ){
        if ( dsu.Find(i) == i ){
            conCmp++;
        }
    }
    cout << conCmp << '\n';
    return 0 ;
}
```

## DSU CODE (Merge in a range )

1. [C. Restructuring Company](https://codeforces.com/edu/course/2/lesson/7/2/practice/contest/289391/problem/C)

```cpp
#include <bits/stdc++.h>
using namespace std;

const int mx = 2e5+123;
struct Disjoint_set_union{
    int Parent[mx], Size[mx];
    void Init(int n){
        for (int i = 0; i <= n; i++){
            Parent[i] = i;
            Size[i] = 1;
        }
    }
    int Find(int v){
        if (v == Parent[v]) return v;
        return Parent[v] = Find(Parent[v]);
    }
    bool Union(int u, int v){
        int root_u = Find(u);
        int root_v = Find(v);
        if (root_u != root_v){
            if (Size[root_u] < Size[root_v]){
                swap(root_u, root_v);
            }
            Parent[root_v] = root_u;
            Size[root_u] += Size[root_v];
            return true;
        }
        return false;
    }
} dsu;

int nxt[mx];
int getNext(int x ){
    if ( nxt[x] == x ) return x ;
    return nxt[x] = getNext(nxt[x]);
}

void solve() {
    int n , q ;
    cin >> n >> q ;
    dsu.Init(n);
    for ( int i = 1 ; i <= n ; i++ ){
        nxt[i] = i;
    }
    for ( int i = 0 ; i < q ; i++ ){
        int x , u , v ;
        cin >> x >> u >> v ;
        if ( x == 1 ){
            dsu.Union(u , v);
        }
        else if ( x == 2 ){
            int cur = getNext(u);
            while ( cur < v ){
                dsu.Union(cur , cur+1);
                nxt[cur] = getNext(cur+1);
                cur = getNext(cur);
            }
        }
        else{
            if ( dsu.Find(u) != dsu.Find(v)){
                cout << "NO" << '\n';
            }
            else{
                cout << "YES" << '\n';
            }
        }
    }
}

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int tests = 1;
    // cin >> tests;
    for ( int tc = 1 ; tc <= tests ; tc++ ){
        solve();
    }
    return 0;
}
```
