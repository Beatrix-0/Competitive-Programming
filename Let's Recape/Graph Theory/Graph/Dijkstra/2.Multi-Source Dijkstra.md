# Multi-Source Dijkstra (O(E log V))

Multi-source Dijkstra is a variation of Dijkstra’s algorithm where instead of starting from one node, we start from multiple source nodes at the same time. <br>

**Note :** The minimum shortest distance from the nearest source node to every other node.

## Code

```cpp
#include <bits/stdc++.h>
using namespace std;

#ifndef ONLINE_JUDGE
#include "trace.cpp"
#else
#define dbg(...)
#endif
#define int long long

const int mx = 2e5 + 123;
const long long INF = 1e17;
vector<pair<int, int>> g[mx];
int dis[mx];
int n, m;

void dijkstra(vector<int> sources)
{
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

    for (auto s : sources) {
        dis[s] = 0;
        pq.push({0, s});
    }

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dis[u]) continue;

        for (auto [v, w] : g[u]) {
            if (d + w < dis[v]) {
                dis[v] = d + w;
                pq.push({dis[v], v});
            }
        }
    }
}

void solve() {
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v, c;
        cin >> u >> v >> c;
        g[u].push_back({v, c});
        // if undirected:
        // g[v].push_back({u, c});
    }
    fill(dis, dis + n + 1, INF);
    int k ; cin >> k;
    vector<int> sources(k);
    for (int i = 0; i < k; i++) {
        cin >> sources[i];
    }
    dijkstra(sources);

    for (int i = 1; i <= n; i++) {
        cout << dis[i] << " ";
    }
    cout << '\n';
}

int32_t main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int tests = 1;
    // cin >> tests;
    for ( int tc = 1 ; tc <= tests ; tc++ ){
        solve();
    }
    return 0;
}
```
