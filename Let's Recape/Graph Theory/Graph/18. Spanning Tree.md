## What a Spanning Tree does

✅ Keeps all N vertices </br>
✅ Removes edges only </br>
✅ Ends up with exactly N − 1 edges </br>

### Basic C++ code to construct a spanning tree using DFS :

```cpp
#include <bits/stdc++.h>
using namespace std;

const int mx = 1000;
vector<int> g[mx];
vector<pair<int,int>> spanningTreeEdges;
bool vis[mx];

void dfs(int u, int p) {
    vis[u] = true;
    for (int v : g[u]) {
        if (!vis[v]) {
            spanningTreeEdges.push_back({u, v});
            dfs(v, u);
        }
    }
}

int main() {
    int N, M;
    cin >> N >> M;

    for (int i = 0; i < M; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    dfs(1, -1);
    for (auto e : spanningTreeEdges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```
