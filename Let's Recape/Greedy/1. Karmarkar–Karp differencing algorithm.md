## Karmarkar–Karp (KK)

Divide a set of numbers into two subsets such that the sum difference is minimized.

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long

int karmarkarKarp(vector<int>& a) {

    priority_queue<int> pq;
    for (int x : a){
        pq.push(x);
    }

    while (pq.size() > 1) {
        int x = pq.top(); pq.pop();
        int y = pq.top(); pq.pop();
        pq.push(llabs(x - y));
    }
    return pq.top();
}

int main() {
    int n; cin >> n;
    vector<int> v(n);
    for (int i = 0; i < n; i++)
        cin >> v[i];

    int result = karmarkarKarp(a);
    cout << result << endl;

    return 0;
}
```
